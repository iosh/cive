# signTransaction (Local Account)

Signs a transaction with the Account's private key.

## Usage

```ts twoslash
import { mainnet } from "cive/chains";
import { parseCFX } from "cive/utils";
import { privateKeyToAccount } from "cive/accounts";

const account = privateKeyToAccount("0x...", { networkId: mainnet.id });

const signature = await account.signTransaction({
  chainId: 1,
  maxFeePerGas: parseCFX("20"),
  maxPriorityFeePerGas: parseCFX("3"),
  gas: 21000n,
  nonce: 69,
  to: account.address,
});
```
