# getCollateralInfo

## Usage

:::code-group

```ts twoslash [example.ts]
import { publicClient } from "./client";

const storageCollateral = await publicClient.getCollateralInfo();
```

```ts twoslash [client.ts] filename="client.ts"
import { createPublicClient, http } from "cive";
import { mainnet } from "cive/chains";

export const publicClient = createPublicClient({
  chain: mainnet,
  transport: http(),
});
```

:::

## Parameters

### epochTag (optional)

- **Type:** `latest_state`| `latest_confirmed`| `latest_checkpoint` | `earliest`
- **Defaut:** `latest_state`

```ts
const storageCollateral = await publicClient.getCollateralInfo({
  epochTag: "latest_state", // [!code focus]
});
```

### epochNumber (optional)

- **Type:** `bigint`

```ts
const storageCollateral = await publicClient.getCollateralInfo({
  epochNumber: 1n, // [!code focus]
});
```
